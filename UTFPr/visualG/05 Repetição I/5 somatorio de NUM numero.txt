Var
   // Seção de Declarações das variáveis
   cont, soma, num : inteiro

Inicio
   // Seção de Comandos, procedimento, funções, operadores, etc...
   cont <- 1
   soma <- 0
   escreva("Escreva um número: ")
   leia(num)
   enquanto (cont <= num) faca
            escreva(" + ", cont)
            soma <- soma + cont
            cont <- cont + 1
   fimenquanto
   escreva(" = ", soma)
